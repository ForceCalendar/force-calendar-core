Integration Example

import { Calendar, EventStore } from '@lightning/calendar-core';

// Initialize with type safety
const calendar = new Calendar({
    view: 'month',
    weekStartsOn: 0,
    businessHours: { start: '09:00', end: '17:00' }
});

// Add event with full features
const event = calendar.addEvent({
    title: 'Team Meeting',
    start: new Date('2024-12-25T10:00:00'),
    end: new Date('2024-12-25T11:00:00'),
    attendees: [
        { name: 'John Doe', email: 'john@example.com' }
    ],
    reminders: [
        { method: 'email', minutesBefore: 15 }
    ],
    categories: ['meeting', 'important']
});

// Check for conflicts
const conflicts = calendar.checkConflicts(event);
if (conflicts.hasConflicts) {
    console.log(`Found ${conflicts.totalConflicts} conflicts`);
}
